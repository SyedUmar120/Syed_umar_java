public class Stack {

    Node top;

    class Node {
        int data;
        Node next;

        Node(int data) {
            this.data = data;
            this.next = null;
        }
    }

    public void push(int data) {
        Node newNode = new Node(data);
        newNode.next = top;
        top = newNode;
    }

    public void pop() {
        if (top == null) {
            System.out.println("Stack empty");
        } else {
            System.out.println("Popped element: " + top.data);
            top = top.next;
        }
    }

    public void display() {
        if (top == null) {
            System.out.println("Stack empty");
        } else {
            Node temp = top;
            System.out.println("Stack elements:");
            while (temp != null) {
                System.out.println(temp.data);
                temp = temp.next;
            }
        }
    }
    public static void main(String[] args) {

        Stack s = new Stack();
        s.push(4);
        s.push(8);
        s.push(5);
        s.pop();
        s.display();
    }
}
